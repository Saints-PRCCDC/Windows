	wevtutil sl "Microsoft-Windows-PowerShell/Operational" /ms:1024000

	setx /M MP_FORCE_USE_SANDBOX 1

"%ProgramFiles%"\"Windows Defender"\MpCmdRun.exe -SignatureUpdate

12. reboot computer
